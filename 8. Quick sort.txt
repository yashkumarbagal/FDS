def partition(array,low,high):
    pivot=array[low]
    i=low+1
    j=high
    flag=False
    while(not flag):
        while(i<=j and array[i]<=pivot):
            i=i+1
        while(i<=j and array[j]>=pivot):
            j=j-1;
        if(j<i):
            flag=True
        else:
            temp=array[low]
            array[i]=array[j]
            array[j]=temp
    temp=array[low]
    array[low]=array[j]
    array[j]=temp    
    return j
def Quick(array,low,high):
    if (low<high):
        m = partition(array,low,high)
        Quick(array,low,m-1)
        Quick(array,m+1,high)
